\documentclass[a4paper]{article}  % Specifies the type of document you're writing (an article) and the paper size (A4).

\title{Competitive Programming Note}  % Sets the title of your document.
\author{Moon Jam}  % Sets the author of your document.
\date{\today}

\usepackage[margin=1in]{geometry}  % Loads the geometry package, which allows you to easily change the margins of your document. Here, you're setting all margins to 1 inch.
\usepackage{amsmath}  % Loads the amsmath package, which provides many useful features for typesetting math.
\usepackage{fancyhdr}  % Loads the fancyhdr package, which allows you to customize the headers and footers of your document.
\usepackage{graphicx}  % Loads the graphicx package, which allows you to include graphics in your document.
\usepackage{cancel}  % Loads the cancel package, which provides commands for striking out terms in math mode.
\usepackage{titlesec}  % Loads the titlesec package, which allows you to customize the formatting of section titles.
\usepackage{titling}  % Loads the titling package, which allows you to customize the title of your document.
\usepackage[hidelinks]{hyperref}

% Set up fancy header/footer
\pagestyle{fancy}  % Changes the page style to fancy, which allows you to use the fancyhdr package to customize your headers and footers.
\fancyhead[LO,L]{Moon Jam}  % Sets the left part of the header to "Moon Jam".
\fancyhead[CO,C]{Competitive Programming Note}  % Sets the center part of the header to "Competitive Programming Note".
\fancyhead[RO,R]{\today}  % Sets the right part of the header to the current date.
\fancyfoot[LO,L]{}  % Leaves the left part of the footer empty.
\fancyfoot[CO,C]{\thepage}  % Sets the center part of the footer to the current page number.
\fancyfoot[RO,R]{}  % Leaves the right part of the footer empty.
\renewcommand{\headrulewidth}{0.4pt}  % Changes the thickness of the horizontal line under the header.
\renewcommand{\footrulewidth}{0.4pt}  % Changes the thickness of the horizontal line above the footer.

% Customize the title
\pretitle{\begin{center}\LARGE\bfseries\vspace*{\fill}}  % Before the title, start a centered environment, make the text large and bold, and add vertical space to center the title vertically.
\posttitle{\end{center}\vskip 0.5em\vfill}  % After the title, end the centered environment, add a small vertical space, and add more vertical space to keep the title centered.

\pagenumbering{arabic}
